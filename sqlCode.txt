/* =========================================================
   ESTATE
   ========================================================= */
CREATE TABLE estate (
    estate_id SERIAL PRIMARY KEY,
    estate_name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);


/* =========================================================
   DIVISION
   ========================================================= */
CREATE TABLE division (
    division_id SERIAL PRIMARY KEY,
    estate_id INT NOT NULL,
    division_name VARCHAR(50) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),

    CONSTRAINT fk_division_estate
    FOREIGN KEY (estate_id)
    REFERENCES estate (estate_id)
    ON DELETE CASCADE
);


/* =========================================================
   FIELD (NC01, 14A, etc.)
   ========================================================= */
CREATE TABLE field (
    field_id SERIAL PRIMARY KEY,
    division_id INT NOT NULL,
    field_code VARCHAR(20) NOT NULL,
    extent NUMERIC(6,2),
    soil_type VARCHAR(50),
    slope VARCHAR(50),
    health VARCHAR(50),

    CONSTRAINT fk_field_division
    FOREIGN KEY (division_id)
    REFERENCES division (division_id)
    ON DELETE CASCADE
);


/* =========================================================
   DIVISION CROP DETAILS (MONTHLY)
   ========================================================= */
CREATE TABLE division_crop_details (
    crop_id SERIAL PRIMARY KEY,
    estate_id INT NOT NULL,
    division_id INT NOT NULL,
    year INT NOT NULL,
    month VARCHAR(15) NOT NULL,

    green_leaf NUMERIC,
    pluckers INT,
    cash_kilo NUMERIC,
    without_cash_avg NUMERIC,

    CONSTRAINT fk_crop_estate
    FOREIGN KEY (estate_id)
    REFERENCES estate (estate_id)
    ON DELETE CASCADE,

    CONSTRAINT fk_crop_division
    FOREIGN KEY (division_id)
    REFERENCES division (division_id)
    ON DELETE CASCADE,

    CONSTRAINT uq_division_year_month
    UNIQUE (division_id, year, month)
);


/* =========================================================
   FERTILIZER APPLICATION (FIELD LEVEL)
   ========================================================= */
CREATE TABLE fertilizer_application (
    application_id SERIAL PRIMARY KEY,
    estate_id INT NOT NULL,
    division_id INT NOT NULL,
    field_id INT NOT NULL,

    year INT NOT NULL,
    month VARCHAR(15),

    fertilizer_type VARCHAR(50) NOT NULL,   -- e.g. U 709
    yph_month INT,
    yph_todate INT,
    quantity_kg NUMERIC,
    last_manure_date DATE,

    CONSTRAINT fk_fert_estate
    FOREIGN KEY (estate_id)
    REFERENCES estate (estate_id)
    ON DELETE CASCADE,

    CONSTRAINT fk_fert_division
    FOREIGN KEY (division_id)
    REFERENCES division (division_id)
    ON DELETE CASCADE,

    CONSTRAINT fk_fert_field
    FOREIGN KEY (field_id)
    REFERENCES field (field_id)
    ON DELETE CASCADE
);


/* =========================================================
   FINANCIAL PERFORMANCE (PROFIT & LOSS)
   ========================================================= */
CREATE TABLE financial_performance (
    finance_id SERIAL PRIMARY KEY,
    estate_id INT NOT NULL,
    year INT NOT NULL,

    made_tea_kg NUMERIC,
    yield NUMERIC,
    nsa NUMERIC,
    cop NUMERIC,
    plucking_cost NUMERIC,
    sundry_income NUMERIC,
    rainfall NUMERIC,

    profit_loss NUMERIC,
    profit_loss_with_sundry NUMERIC,

    CONSTRAINT fk_finance_estate
    FOREIGN KEY (estate_id)
    REFERENCES estate (estate_id)
    ON DELETE CASCADE,

    CONSTRAINT uq_estate_year
    UNIQUE (estate_id, year)
);
